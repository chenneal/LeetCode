My main idea

  • sort the list with pool_id && start_time.
  • with the same pool_id, record the start_max && end_min, if cur_start > end_min, restart the process at the current position.
  • do the same process with other pool_id
  • loop
#include <iostream>
#include <vector>
#include <stdlib.h>

using namespace std;

// define the comparation
int compare(const void *a, const void *b) {
    long long* x = (long long *)a;
    long long* y = (long long *)b; 
    if (x[1] > y[1])   return 1;
    else if (x[1] == y[1] && x[2] > y[2])   return 1;
    else return -1; 
}

long long select_max_concurrent(long long **data, int N) {
    // record the current pool id
    long long pool_id;
    // record the max number of concurrent
    int max = 0;
    // record the index of the connection
    int count = 0;
    // record the max-concurrent pool id
    long long max_pool_id;
    while (count < N) {
        // record the max-concurrent number of current pool_id 
        int max_pool = 0;
        // record the current concurrent number of current pool_id 
        int cur_max_pool = 0;
        long long begin_max = 0;
        long long end_min = LLONG_MAX;
        pool_id = data[count][1];
        while (count < N) {
            // if begin_next > end_min, restart the current concurrent count
            if (data[count][2] >= end_min) {
                begin_max = 0;
                end_min = LLONG_MAX;
                count++;
                cur_max_pool = 0;
            }
            else {
                if (data[count][2] > begin_max)   begin_max = data[count][2];
                if (data[count][3] < end_min)     end_min = data[count][3];
                cur_max_pool++;
                if (cur_max_pool > max_pool) {
                    max_pool = cur_max_pool;
                }
                count++;
            }
            if (count < N) {
                if (pool_id != data[count][1]) {
                    break;
                }
            }
            if (cur_max_pool > max_pool) {
                max_pool = cur_max_pool;
            }
        }
        if (max_pool > max) {
            max = max_pool;
            max_pool_id = pool_id;
        }
    }
    return max_pool_id; 
}

int main (void) {
    int N;
    long long **data;
    long long * p;
    cout << "enter the connection number\n";
    cin >> N;
    data = (long long **) malloc (N*sizeof(long long*));
    p = (long long *) malloc (4*N*sizeof(long long));

    for (int i = 0; i < N; i++) {
        data[i] = p;
        p = p + 4;
    }

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < 4; j++) {
            cin >> data[i][j];
        }
    }

    // sort the array with the key pool_id && the start_time
    qsort(data[0], N, 4*sizeof(long long), compare);

    long long max_pool_id = select_max_concurrent(data, N);
    // return the max_concurrent_pool_id;
    cout << "max concurrent pool id = " << max_pool_id << endl;
    return 0;
}
